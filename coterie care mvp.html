        window.onclick = function(event) {

                event.target.style.display = 'none';

            }

        }


        // Track scroll depth and engagement

        let maxScrollDepth = 0;

        window.addEventListener('scroll', function() {

            const scrollPercent = Math.round((window.scrollY / (document.body.scrollHeight - window.innerHeight)) * 100);

            if (scrollPercent > maxScrollDepth) {

                maxScrollDepth = scrollPercent;

                userSession.scrollDepth = maxScrollDepth;

                

                // Track milestone scroll depths

                if (scrollPercent >= 25 && maxScrollDepth < 30) {

                    trackInteraction('scroll_depth', '25_percent');

                } else if (scrollPercent >= 50 && maxScrollDepth < 55) {

                    trackInteraction('scroll_depth', '50_percent');

                } else if (scrollPercent >= 75 && maxScrollDepth < 80) {

                    trackInteraction('scroll_depth', '75_percent');

                } else if (scrollPercent >= 90) {

                    trackInteraction('scroll_depth', '90_percent');

                }

            }

            

            const cards = document.querySelectorAll('.feature-card');

            cards.forEach(card => {

                const rect = card.getBoundingClientRect();

                if (rect.top < window.innerHeight && rect.bottom > 0) {

                    card.style.transform = 'translateY(0)';

                    card.style.opacity = '1';

                }

            });

        });

                    event_category: 'engagement',

                });

                    event_category: 'engagement',

                    event_label: 'max_scroll',

            }
